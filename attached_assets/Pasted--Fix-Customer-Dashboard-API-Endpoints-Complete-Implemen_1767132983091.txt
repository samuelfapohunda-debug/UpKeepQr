## Implementation Guide for Replit Agent

### Step 1: Create Customer API Routes File

**File:** `server/src/routes/customer.ts`

```typescript
import { Router } from "express";
import { db } from "@db";
import { households, householdTaskAssignments, homeMaintenanceTasks } from "@shared/schema";
import { eq } from "drizzle-orm";

const router = Router();

// GET /api/households/:id
// Get household information for customer dashboard
router.get("/:householdId", async (req, res) => {
  try {
    const { householdId } = req.params;

    console.log('📊 Fetching household data for:', householdId);

    const [household] = await db
      .select({
        id: households.id,
        firstName: households.firstName,
        lastName: households.lastName,
        email: households.email,
        phone: households.phone,
        homeType: households.homeType,
        city: households.city,
        state: households.state,
        zipCode: households.zipCode,
        squareFootage: households.squareFootage,
        hvacType: households.hvacType,
        waterHeaterType: households.waterHeaterType,
        setupStatus: households.setupStatus,
      })
      .from(households)
      .where(eq(households.id, householdId));

    if (!household) {
      console.warn('⚠️  Household not found:', householdId);
      return res.status(404).json({ error: "Household not found" });
    }

    console.log('✅ Household data found:', household.email);
    res.json(household);
  } catch (error) {
    console.error("❌ Error fetching household:", error);
    res.status(500).json({ error: "Failed to fetch household data" });
  }
});

// GET /api/households/:id/tasks
// Get all tasks for a household with full details
router.get("/:householdId/tasks", async (req, res) => {
  try {
    const { householdId } = req.params;

    console.log('📋 Fetching tasks for household:', householdId);

    // Join task assignments with task catalog to get full details
    const tasks = await db
      .select({
        id: householdTaskAssignments.id,
        taskId: householdTaskAssignments.taskId,
        taskName: homeMaintenanceTasks.taskName,
        description: homeMaintenanceTasks.description,
        category: homeMaintenanceTasks.category,
        priority: householdTaskAssignments.priority,
        status: householdTaskAssignments.status,
        dueDate: householdTaskAssignments.dueDate,
        frequency: householdTaskAssignments.frequency,
        completedAt: householdTaskAssignments.completedAt,
        notes: householdTaskAssignments.notes,
        estimatedDuration: homeMaintenanceTasks.estimatedDuration,
        diyDifficulty: homeMaintenanceTasks.diyDifficulty,
        seasonalTiming: homeMaintenanceTasks.seasonalTiming,
      })
      .from(householdTaskAssignments)
      .leftJoin(
        homeMaintenanceTasks,
        eq(householdTaskAssignments.taskId, homeMaintenanceTasks.id)
      )
      .where(eq(householdTaskAssignments.householdId, householdId))
      .orderBy(householdTaskAssignments.dueDate);

    console.log(`✅ Found ${tasks.length} tasks for household`);

    // Map to camelCase for frontend consistency
    const formattedTasks = tasks.map(task => ({
      id: task.id,
      taskId: task.taskId,
      taskName: task.taskName,
      description: task.description,
      category: task.category,
      priority: task.priority,
      status: task.status,
      dueDate: task.dueDate,
      frequency: task.frequency,
      completedAt: task.completedAt,
      notes: task.notes,
      estimatedDuration: task.estimatedDuration,
      diyDifficulty: task.diyDifficulty,
      seasonalTiming: task.seasonalTiming,
    }));

    res.json(formattedTasks);
  } catch (error) {
    console.error("❌ Error fetching tasks:", error);
    res.status(500).json({ error: "Failed to fetch tasks" });
  }
});

export default router;
```

### Step 2: Register Routes in Main Server File

**File:** `server/index.ts`

Add these lines:

```typescript
// Add import at top
import customerRouter from "./src/routes/customer";

// Add route registration (after other routes)
app.use("/api/households", customerRouter);

// IMPORTANT: This should be BEFORE the catch-all route that serves index.html
// Make sure your route registration looks like this:

// API routes (these must come FIRST)
app.use("/api/admin/households", appliancesRouter);
app.use("/api/admin/households", householdsRouter);
app.use("/api/households", customerRouter);  // NEW - Add this line
// ... other API routes

// Frontend route handler (this must come LAST)
app.get("*", (_req, res) => {
  res.sendFile(join(process.cwd(), "dist", "public", "index.html"));
});
```

**⚠️ CRITICAL:** The customer routes MUST be registered BEFORE the `app.get("*", ...)` catch-all route, otherwise the API will return HTML instead of JSON.

## Troubleshooting

### Issue: Still getting HTML instead of JSON

**Check route order in server/index.ts:**
```typescript
// ✅ CORRECT ORDER:
app.use("/api/households", customerRouter);  // API routes FIRST
app.get("*", (req, res) => {                 // Catch-all LAST
  res.sendFile(...)
});

// ❌ WRONG ORDER:
app.get("*", (req, res) => {                 // Catch-all FIRST
  res.sendFile(...)
});
app.use("/api/households", customerRouter);  // API routes LAST (won't work!)
```

### Issue: Database query errors

**Check column names match schema:**
```typescript
// In customer.ts, make sure you're using the Drizzle schema:
import { households } from "@shared/schema";

// Use schema column names (camelCase):
households.firstName  // ✅ Correct
households.first_name // ❌ Wrong
```

### Issue: 404 on tasks endpoint

**Verify task assignments exist:**
```sql
SELECT COUNT(*) FROM household_task_assignments 
WHERE household_id = 'your-household-id';
```

If count is 0, tasks haven't been generated for this household yet.
