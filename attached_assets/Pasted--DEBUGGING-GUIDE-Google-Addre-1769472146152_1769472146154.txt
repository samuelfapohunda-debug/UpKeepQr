```tsx
import { useRef, useEffect, useState } from 'react';

export default function SetupForm() {
  const addressRef = useRef(null);
  const [formData, setFormData] = useState({
    streetAddress: '',
    city: '',
    state: '',
    zipCode: ''
  });

  // Initialize Google Places Autocomplete
  useEffect(() => {
    if (!addressRef.current || !window.google) return;

    const autocomplete = new google.maps.places.Autocomplete(
      addressRef.current,
      {
        types: ['address'],
        componentRestrictions: { country: ['us', 'ca'] }
      }
    );

    autocomplete.addListener('place_changed', () => {
      const place = autocomplete.getPlace();
      
      if (!place.address_components) return;

      // Parse address components
      let street = '';
      let city = '';
      let state = '';
      let zip = '';

      place.address_components.forEach((component) => {
        const types = component.types;
        
        if (types.includes('street_number')) {
          street = component.long_name;
        }
        if (types.includes('route')) {
          street += ' ' + component.long_name;
        }
        if (types.includes('locality')) {
          city = component.long_name;
        }
        if (types.includes('administrative_area_level_1')) {
          state = component.short_name;
        }
        if (types.includes('postal_code')) {
          zip = component.long_name;
        }
      });

      // Update form state
      setFormData({
        ...formData,
        streetAddress: street.trim(),
        city: city,
        state: state,
        zipCode: zip
      });
    });
  }, []);

  return (
    <form>
      {/* Street Address - WITH AUTOCOMPLETE */}
      <div className="mb-5">
        <label 
          htmlFor="streetAddress" 
          className="block text-sm font-semibold text-gray-700 mb-2"
        >
          Street Address *
        </label>
        <input
          ref={addressRef}  // â† CRITICAL: Google Places attaches here
          id="streetAddress"
          type="text"
          name="streetAddress"
          value={formData.streetAddress}
          onChange={(e) => setFormData({...formData, streetAddress: e.target.value})}
          required
          className="w-full bg-white border border-gray-300 rounded-md px-4 py-3 text-base text-gray-900 focus:outline-none focus:ring-2 focus:ring-blue-500"
          placeholder=""
        />
        <p className="text-sm text-gray-600 mt-2">
          Start typing for suggestions (US & Canada)
        </p>
      </div>

      {/* City - AUTO-FILLED */}
      <div className="mb-5">
        <label 
          htmlFor="city" 
          className="block text-sm font-semibold text-gray-700 mb-2"
        >
          City
        </label>
        <input
          id="city"
          type="text"
          name="city"
          value={formData.city}
          disabled
          className="w-full bg-white border border-gray-300 rounded-md px-4 py-3 text-base text-gray-500 cursor-not-allowed"
        />
        <p className="text-sm text-gray-600 mt-2">
          Auto-fills from address
        </p>
      </div>

      {/* State - AUTO-FILLED */}
      <div className="mb-5">
        <label 
          htmlFor="state" 
          className="block text-sm font-semibold text-gray-700 mb-2"
        >
          State / Province
        </label>
        <input
          id="state"
          type="text"
          name="state"
          value={formData.state}
          disabled
          className="w-full bg-white border border-gray-300 rounded-md px-4 py-3 text-base text-gray-500 cursor-not-allowed"
        />
        <p className="text-sm text-gray-600 mt-2">
          Auto-fills from address
        </p>
      </div>

      {/* ZIP - AUTO-FILLED */}
      <div className="mb-5">
        <label 
          htmlFor="zipCode" 
          className="block text-sm font-semibold text-gray-700 mb-2"
        >
          ZIP / Postal Code *
        </label>
        <input
          id="zipCode"
          type="text"
          name="zipCode"
          value={formData.zipCode}
          onChange={(e) => setFormData({...formData, zipCode: e.target.value})}
          required
          className="w-full bg-white border border-gray-300 rounded-md px-4 py-3 text-base text-gray-900 focus:outline-none focus:ring-2 focus:ring-blue-500"
        />
        <p className="text-sm text-gray-600 mt-2">
          For weather-based reminders and local service matching
        </p>
      </div>
    </form>
  );
}
```

```tsx
  document.head.appendChild(script);
}, []);
```
