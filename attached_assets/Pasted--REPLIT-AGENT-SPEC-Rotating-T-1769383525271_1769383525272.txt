import { useState, useEffect } from 'react';

export default function Home() {
  const statistics = [
    "Trusted by 5000+ Homeowner subscriptions in North America",
    "Trusted by 100+ Realtor subscriptions in North America",
    "Trusted by 20+ Apartment Complex subscriptions in North America"
  ];

  const [currentIndex, setCurrentIndex] = useState(0);
  const [isAnimating, setIsAnimating] = useState(false);

  useEffect(() => {
    const interval = setInterval(() => {
      setIsAnimating(true);
      
      setTimeout(() => {
        setCurrentIndex((prevIndex) => (prevIndex + 1) % statistics.length);
        setIsAnimating(false);
      }, 500); // Half of animation duration
      
    }, 4000); // 4 seconds display time

    return () => clearInterval(interval);
  }, []);

  // Helper function to format text with bold numbers
  const formatStatistic = (text: string) => {
    // Match patterns like "5000+", "100+", "20+"
    const parts = text.split(/(\d+\+)/);
    
    return parts.map((part, index) => {
      if (/\d+\+/.test(part)) {
        return <strong key={index} className="font-bold text-gray-900">{part}</strong>;
      }
      return <span key={index}>{part}</span>;
    });
  };

  return (
    <div className="homepage">
      {/* Hero Section */}
      <section className="hero">
        {/* Existing hero content */}
      </section>

      {/* Rotating Trust Badge Section */}
      <section className="trust-badge-section">
        <div className="trust-badge-container">
          <p 
            className={`trust-badge-text ${isAnimating ? 'rotating-out' : 'rotating-in'}`}
            key={currentIndex}
          >
            {formatStatistic(statistics[currentIndex])}
          </p>
        </div>
      </section>

      {/* Rest of homepage */}
    </div>
  );
}

export default function Home() {
  const statistics = [
    { text: "Trusted by", number: "5000+", rest: "Homeowner subscriptions in North America" },
    { text: "Trusted by", number: "100+", rest: "Realtor subscriptions in North America" },
    { text: "Trusted by", number: "20+", rest: "Apartment Complex subscriptions in North America" }
  ];

  const [currentIndex, setCurrentIndex] = useState(0);
  const [animationClass, setAnimationClass] = useState('animate-rotate-in');

  useEffect(() => {
    const interval = setInterval(() => {
      setAnimationClass('animate-rotate-out');
      
      setTimeout(() => {
        setCurrentIndex((prev) => (prev + 1) % statistics.length);
        setAnimationClass('animate-rotate-in');
      }, 500);
      
    }, 4000);

    return () => clearInterval(interval);
  }, []);

  const current = statistics[currentIndex];

  return (
    <div className="w-full bg-gray-50 py-10 flex justify-center items-center">
      <div className="relative h-16 max-w-4xl overflow-hidden flex items-center justify-center">
        <p 
          className={`text-lg text-gray-600 text-center absolute ${animationClass}`}
          key={currentIndex}
        >
          {current.text} <span className="font-bold text-xl text-gray-900">{current.number}</span> {current.rest}
        </p>
      </div>
    </div>
  );
}
