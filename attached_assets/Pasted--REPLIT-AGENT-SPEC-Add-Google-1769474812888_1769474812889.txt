Step 4: Initialize Google Places Autocomplete
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

Add useEffect to initialize autocomplete ONLY for Request a Pro form:

```tsx
useEffect(() => {
  // Check if Google Maps is loaded
  if (!requestProAddressRef.current || !window.google) {
    console.log('Google Maps not loaded yet');
    return;
  }

  console.log('Initializing autocomplete for Request a Pro form');

  // Initialize autocomplete on Street Address field
  const autocomplete = new google.maps.places.Autocomplete(
    requestProAddressRef.current,
    {
      types: ['address'],
      componentRestrictions: { country: ['us', 'ca'] } // US & Canada
    }
  );

  // Listen for place selection
  autocomplete.addListener('place_changed', () => {
    const place = autocomplete.getPlace();
    
    console.log('Place selected:', place);

    if (!place.address_components) {
      console.log('No address components found');
      return;
    }

    // Parse address components
    let street = '';
    let city = '';
    let state = '';
    let zip = '';

    place.address_components.forEach((component) => {
      const types = component.types;
      
      // Street number (e.g., "123")
      if (types.includes('street_number')) {
        street = component.long_name;
      }
      
      // Street name (e.g., "Main St")
      if (types.includes('route')) {
        street += (street ? ' ' : '') + component.long_name;
      }
      
      // City (e.g., "Atlanta")
      if (types.includes('locality')) {
        city = component.long_name;
      }
      
      // State (e.g., "GA")
      if (types.includes('administrative_area_level_1')) {
        state = component.short_name; // Use short_name for "GA" not "Georgia"
      }
      
      // ZIP Code (e.g., "30301")
      if (types.includes('postal_code')) {
        zip = component.long_name;
      }
    });

    console.log('Parsed address:', { street, city, state, zip });

    // Update form state with parsed address
    setFormData(prev => ({
      ...prev,
      streetAddress: street.trim(),
      city: city,
      state: state,
      zipCode: zip
    }));
  });

  // Cleanup function (optional but good practice)
  return () => {
    google.maps.event.clearInstanceListeners(autocomplete);
  };
}, []); // Empty dependency array - run once on mount
```
