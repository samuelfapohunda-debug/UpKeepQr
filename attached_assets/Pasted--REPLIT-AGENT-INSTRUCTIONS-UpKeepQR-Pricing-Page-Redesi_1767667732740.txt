### Step 3: Pricing Card Component

**Create reusable component:** `<PricingCard />`
```tsx
interface PricingCardProps {
  title: string;
  price: string;
  billedAmount: string;
  badge?: string;
  features: string[];
  ctaLabel: string;
  ctaLink: string;
  highlighted?: boolean;
}

function PricingCard({
  title,
  price,
  billedAmount,
  badge,
  features,
  ctaLabel,
  ctaLink,
  highlighted = false
}: PricingCardProps) {
  return (
    <div className={`
      relative bg-white rounded-xl p-8 
      border-2 transition-all duration-300
      hover:shadow-xl hover:-translate-y-1
      ${highlighted 
        ? 'border-emerald-500 shadow-lg' 
        : 'border-slate-200 shadow-sm'
      }
    `}>
      {/* Badge (if exists) */}
      {badge && (
        <div className="absolute -top-4 left-1/2 -translate-x-1/2">
          <span className="bg-emerald-500 text-white px-4 py-1 rounded-full text-sm font-semibold">
            {badge}
          </span>
        </div>
      )}
      
      {/* Title */}
      <h3 className="text-2xl font-bold text-slate-900 mb-2">
        {title}
      </h3>
      
      {/* Pricing */}
      <div className="mb-6">
        <div className="flex items-baseline gap-2">
          <span className="text-4xl font-bold text-slate-900">
            {price}
          </span>
          <span className="text-slate-600">/ month</span>
        </div>
        <p className="text-sm text-slate-600 mt-1">
          {billedAmount}
        </p>
      </div>
      
      {/* Features List */}
      <ul className="space-y-3 mb-8">
        {features.map((feature, index) => (
          <li key={index} className="flex items-start gap-3">
            <svg className="w-5 h-5 text-emerald-500 flex-shrink-0 mt-0.5" fill="currentColor" viewBox="0 0 20 20">
              <path fillRule="evenodd" d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z" clipRule="evenodd" />
            </svg>
            <span className="text-slate-700">{feature}</span>
          </li>
        ))}
      </ul>
      
      {/* CTA Button */}
      
        href={ctaLink}
        className={`
          block w-full text-center py-3 px-6 rounded-lg font-semibold
          transition-all duration-200
          ${highlighted
            ? 'bg-emerald-600 text-white hover:bg-emerald-700 hover:shadow-lg'
            : 'bg-white text-emerald-600 border-2 border-emerald-600 hover:bg-emerald-50'
          }
        `}
      >
        {ctaLabel}
      </a>
    </div>
  );
}
```
