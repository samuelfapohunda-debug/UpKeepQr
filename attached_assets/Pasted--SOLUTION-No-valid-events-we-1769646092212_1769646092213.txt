function formatICSDate(date: Date): string {
  if (!date || isNaN(date.getTime())) {
    console.error('Invalid date:', date);
    return formatICSDate(new Date()); // Fallback to today
  }
  
  const year = date.getFullYear();
  const month = String(date.getMonth() + 1).padStart(2, '0');
  const day = String(date.getDate()).padStart(2, '0');

}

// Generate .ics file content
function generateICSFile(tasks: any[], householdName: string): string {
  console.log('Generating .ics file for', tasks.length, 'tasks');
  
  const now = new Date();
  const timestamp = formatICSDateTime(now);
  
  // Start calendar
  let ics = `BEGIN:VCALENDAR
VERSION:2.0
PRODID:-//UpKeepQR//Home Maintenance Tasks//EN
CALSCALE:GREGORIAN
METHOD:PUBLISH
X-WR-CALNAME:UpKeepQR - ${escapeICS(householdName)}
X-WR-TIMEZONE:America/New_York
X-WR-CALDESC:Home maintenance tasks for ${escapeICS(householdName)}
BEGIN:VTIMEZONE
TZID:America/New_York
TZOFFSETFROM:-0500
TZOFFSETTO:-0400
TZNAME:EDT
DTSTART:19700308T020000
RRULE:FREQ=YEARLY;BYMONTH=3;BYDAY=2SU
END:VTIMEZONE\n`;

  // Add each task as event
  let validEventCount = 0;
  
  tasks.forEach((task, index) => {
    try {
      // Validate required fields
      if (!task.id) {
        console.warn(`Task ${index}: Missing ID, skipping`);
        return;
      }
      
      if (!task.taskName || task.taskName.trim() === '') {
        console.warn(`Task ${index}: Empty taskName, skipping`);
        return;
      }
      
      if (!task.dueDate) {
        console.warn(`Task ${index}: Missing dueDate, skipping`);
        return;
      }

      // Generate event
      const uid = `task-${task.id}@upkeepqr.com`;
      const dueDate = formatICSDate(new Date(task.dueDate));
      const priority = task.priority === 'High' ? '1' : 
                       task.priority === 'Medium' ? '5' : '9';
      
      // Build description (use howTo, not description)
      let description = task.howTo || 'Home maintenance task';
      description += `\\n\\nPriority: ${task.priority || 'Medium'}`;
      description += `\\n\\nCategory: ${task.category || 'Maintenance'}`;
      description += `\\n\\nFrom UpKeepQR`;
      
      // Add event
      ics += `BEGIN:VEVENT
UID:${uid}
DTSTAMP:${timestamp}
DTSTART;VALUE=DATE:${dueDate}
DTEND;VALUE=DATE:${dueDate}
SUMMARY:${escapeICS(task.taskName)}
DESCRIPTION:${escapeICS(description)}
PRIORITY:${priority}
CATEGORIES:${escapeICS(task.category || 'Maintenance')},Home Maintenance
STATUS:NEEDS-ACTION
CLASS:PUBLIC
TRANSP:TRANSPARENT
LOCATION:${escapeICS(householdName)}
BEGIN:VALARM
TRIGGER:-P1D
DESCRIPTION:Reminder: ${escapeICS(task.taskName)} is due tomorrow
ACTION:DISPLAY
END:VALARM
BEGIN:VALARM
TRIGGER:-P7D
DESCRIPTION:Reminder: ${escapeICS(task.taskName)} is due in 1 week
ACTION:DISPLAY
END:VALARM
END:VEVENT\n`;

      validEventCount++;
      console.log(`Task ${index + 1}: "${task.taskName}" - Added âœ“`);
      
    } catch (error) {
      console.error(`Error generating event ${index}:`, error);
    }
  });

  // End calendar
  ics += `END:VCALENDAR`;
  
  console.log(`Generated ${validEventCount} valid events`);
  
  if (validEventCount === 0) {
    console.error('No valid events generated!');
    return '';
  }
  
  return ics;
}

// Format date as YYYYMMDD
function formatICSDate(date: Date): string {
  if (!date || isNaN(date.getTime())) {
    console.error('Invalid date, using today:', date);
    date = new Date();
  }
  
  const year = date.getFullYear();
  const month = String(date.getMonth() + 1).padStart(2, '0');
  const day = String(date.getDate()).padStart(2, '0');
  return `${year}${month}${day}`;
}

// Format datetime as YYYYMMDDTHHmmssZ
function formatICSDateTime(date: Date): string {
  if (!date || isNaN(date.getTime())) {
    date = new Date();
  }
  
  const year = date.getUTCFullYear();
  const month = String(date.getUTCMonth() + 1).padStart(2, '0');
  const day = String(date.getUTCDate()).padStart(2, '0');
  const hours = String(date.getUTCHours()).padStart(2, '0');
  const minutes = String(date.getUTCMinutes()).padStart(2, '0');
  const seconds = String(date.getUTCSeconds()).padStart(2, '0');
  return `${year}${month}${day}T${hours}${minutes}${seconds}Z`;
}

// Escape special characters for .ics
function escapeICS(text: string): string {
  if (!text) return '';
  
  return String(text)
    .replace(/\\/g, '\\\\')    // Backslash FIRST
    .replace(/;/g, '\\;')      // Semicolon
    .replace(/,/g, '\\,')      // Comma  
    .replace(/\n/g, '\\n')     // Newline
    .replace(/\r/g, '')        // Remove CR
    .substring(0, 1000);       // Limit length
}

// Sanitize filename
function sanitizeFilename(filename: string): string {
  return filename
    .replace(/[^a-z0-9]/gi, '_')
    .replace(/_+/g, '_')
    .substring(0, 50);
}
